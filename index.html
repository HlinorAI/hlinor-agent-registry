examples/agent.yaml YAML / registry 
id: web-research-agent
name: Web Research Agent
department: marketing
skills:
  - web_search
  - summarize_text
policies:
  - max_10_searches_per_hour
  - require_budget_check
allowed_actions:
  - search
  - read_public_url
  - extract_keywords
blocked_actions:
  - login_to_website
  - call_premium_paid_api
